# 2. Run `compinit`
# 3. Source plugins

$ZSH path:plugins/git/git.plugin.zsh
$ZSH path:plugins/copybuffer/copybuffer.plugin.zsh # press `^O` to copy buffer

zsh-users/zsh-autosuggestions
