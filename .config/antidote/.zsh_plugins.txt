# !! Note: plugin order matters !! #

# Add plugins that use `$fpath` for their completions
$ZSH_CUSTOM/plugins/homebrew/configurate-zsh-completions.zsh

# NOW, use a plugin that runs compinit for you to initialize your completions
$ZSH_CUSTOM/plugins/ohmyzsh/init-completion-system.zsh

# Before loading Oh My Zsh `lib/completion.zsh`
$ZSH_CUSTOM/plugins/zsh-highlighting/patches/fix-disappeared-highlighting.zsh

$ZSH/tools/check_for_upgrade.sh
